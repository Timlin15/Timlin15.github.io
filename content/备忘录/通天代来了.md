登录自己的账号：
```bash
# codex
mkdir ~/.codex-account-A
CODEX_HOME=~/.codex-account-A codex login --device-auth
alias codex-A='CODEX_HOME=~/.codex-A codex'
echo "alias codex-A='CODEX_HOME=~/.codex-A codex'" >> ~/.bashrc

# claude
mkdir ~/.claude-personal
CLAUDE_CONFIG_DIR=~/.claude-personal claude /login
alias cc-personal='CLAUDE_CONFIG_DIR=~/.claude-personal claude'
echo "alias cc-personal='CLAUDE_CONFIG_DIR=~/.claude-personal claude'" >> ~/.bashrc
```
